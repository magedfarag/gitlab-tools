🛡️ Security Assessment Criteria:

Risk Levels: Critical (Any critical vulns), High (3+ high vulns), Medium (1-2 high vulns), Low (No critical/high vulns)

Security Grades: A-F based on vulnerability count and severity distribution

$(foreach($project in $ProjectReports) { $sastScan = $SecurityScanResults | Where-Object { $_.ProjectId -eq $project.ProjectId -and $_.ScanType -eq "SAST" } | Select-Object -First 1 $scaScan = $SecurityScanResults | Where-Object { $_.ProjectId -eq $project.ProjectId -and $_.ScanType -eq "SCA" } | Select-Object -First 1 $criticalVulns = 0 $highVulns = 0 $mediumVulns = 0 $totalVulns = 0 $lastScanDate = "Never" if ($sastScan) { $criticalVulns += $sastScan.CriticalVulnerabilities $highVulns += $sastScan.HighVulnerabilities $mediumVulns += $sastScan.MediumVulnerabilities $totalVulns += $sastScan.TotalVulnerabilities $lastScanDate = $sastScan.ScanDate.ToString("MM/dd/yyyy") } if ($scaScan) { $criticalVulns += $scaScan.CriticalVulnerabilities $highVulns += $scaScan.HighVulnerabilities $mediumVulns += $scaScan.MediumVulnerabilities $totalVulns += $scaScan.TotalVulnerabilities if ($scaScan.ScanDate -and ($lastScanDate -eq "Never" -or $scaScan.ScanDate -gt $sastScan.ScanDate)) { $lastScanDate = $scaScan.ScanDate.ToString("MM/dd/yyyy") } } $riskLevel = "risk-low" if ($criticalVulns -gt 0) { $riskLevel = "risk-critical" } elseif ($highVulns -gt 2) { $riskLevel = "risk-high" } elseif ($highVulns -gt 0) { $riskLevel = "risk-medium" } $riskText = switch ($riskLevel) { "risk-low" { "Low" } "risk-medium" { "Medium" } "risk-high" { "High" } "risk-critical" { "Critical" } default { "Unknown" } } $sastGrade = if ($sastScan) { $sastScan.SecurityGrade } else { "N/A" } $scaGrade = if ($scaScan) { $scaScan.SecurityGrade } else { "N/A" } @" "@ })
Project Name Critical High Medium Total Risk Level SAST Grade SCA Grade Last Scan
$($project.ProjectName) $criticalVulns $highVulns $mediumVulns $totalVulns $riskText $sastGrade $scaGrade $lastScanDate
